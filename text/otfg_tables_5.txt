-- -- Table structure for table `users`-- CREATE TABLE `users` (  `UserID` int(11) NOT NULL auto_increment,  `UserName` varchar(25) NOT NULL,  `Password` varchar(32) default NULL,  `CookieID` varchar(32) default NULL,  `LoginKey` varchar(32) default NULL,  `LoginExp` int(10) unsigned default NULL,  `LoginIP` varchar(15) NOT NULL,  PRIMARY KEY  (`UserID`)) ENGINE=InnoDB DEFAULT CHARSET=latin1 AUTO_INCREMENT=1 ;
